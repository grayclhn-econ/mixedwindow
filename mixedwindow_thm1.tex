If Assumptions~\ref{a1}~--~\ref{a4} hold then
\begin{equation*}
  \sqrt{P} (\fb - \E \fb^*) \to^d N(0, \sigma^2),
\end{equation*}
with $\sigma^2 = s_1 + 2(s_2 + s_3)$ and
\begin{align*}
    s_1  &= \lim \var(\sqrt{P}\, \fb^*), &
    s_2  &= \lim \cov(\sqrt{P}\, \fb^*, \sqrt{P}\, \gb^*), &
    s_3  &= \lim \var(\sqrt{P}\, \gb^*).
\end{align*}
